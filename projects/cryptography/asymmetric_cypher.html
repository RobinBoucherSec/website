# RSA Implementation
def pgdc(a, b):
    while True:
        temp = a % b
        if temp == 0:
            return b
        a = b
        b = temp

def euclide(b, n):
    n0, b0, t0, t = n, b, 0, 1
    q, r = n0 // b0, n0 % b0
    while r > 0:
        temp = t0 - q*t
        t0, t = t, temp % n
        n0, b0 = b0, r
        q, r = n0 // b0, n0 % b0
    return t if b0 == 1 else None

def crypter(message):
    global e, n
    return pow(message, e, n)

def chiffrer(message, bloc):
    if 10**bloc > n:
        return None
    liste = []
    chiffres = "".join(f"{ord(c):03d}" for c in message)
    while len(chiffres) % bloc != 0:
        chiffres = "0" + chiffres
    for i in range(0, len(chiffres), bloc):
        liste.append(crypter(int(chiffres[i:i+bloc])))
    return liste
