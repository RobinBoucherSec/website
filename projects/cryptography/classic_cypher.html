Caesar & Vigenère Ciphers

Description chiffrement de César: Chaque lettre du message est décalée dans l’alphabet (ex. "A" → "D" avec un décalage de 3).

Implementation of Caesar cipher, Vigenère cipher, and substitution cipher with space encoding.

# Caesar Cipher Implementation
def caesar_encrypt(text, shift):
    encrypted = ""
    for char in text:
        if char.isalpha():
            base = ord('a') if char.islower() else ord('A')
            encrypted += chr((ord(char) - base + shift) % 26 + base)
        else:
            encrypted += char
    return encrypted

# Vigenère Cipher Implementation
def vigenere_encrypt(text, key):
    result = ""
    key = key.lower()
    j = 0
    for char in text:
        if char.isalpha():
            shift = ord(key[j % len(key)]) - ord('a')
            result += caesar_encrypt(char, shift)
            j += 1
        else:
            result += char
    return result

# Advanced Vigenère Cipher
def vigenere_chiffrement_avance(texte, cle):
    resultat = ''
    j = 0
    for i in range(len(texte)):
        if texte[i].isalpha():
            base = 65 if texte[i].isupper() else 97
            k = ord(cle[j % len(cle)].lower()) - 97
            t = (ord(texte[i]) - base + k) % 26
            resultat += chr(t + base)
            j += 1
        else:
            resultat += texte[i]
    return resultat

# Coincidence Index Analysis
def indice_coincidence(texte):
    from collections import Counter
    lettres = [c.lower() for c in texte if c.isalpha()]
    N = len(lettres)
    freqs = Counter(lettres)
    ic = sum(f * (f - 1) for f in freqs.values()) / (N * (N - 1)) if N > 1 else 0
    return ic

Applies Caesar cipher multiple times for enhanced security through iteration.

# Iterated Caesar Cipher
def custom_caesar_encrypt(text, key, iterations):
    print(f"Original text: {text}")
    for i in range(iterations):
        text = caesar_encrypt(text, key)
        print(f"Iteration {i+1}: {text}")
    return text

def custom_caesar_decrypt(text, key, iterations):
    print(f"Encrypted text: {text}")
    for i in range(iterations):
        text = caesar_decrypt(text, key)
        print(f"Decryption {i+1}: {text}")
    return text

# Example Usage
message = "Bonjour CyberTeam"
